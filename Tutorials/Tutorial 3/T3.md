```

public class Q4{
  public static void main(String[] args){
    Scanner s = new scanner(System.in);
    System.out.println("Enter the year; ");
    int year = s.nextInt()
    if ( year%4 == 0 && year%100 != 0 || year%400 == 0)
    {
      System.out.println("A Leap Year");
    }
    else {
      System.out.println("Not a Leap Year");
        }
  }
}
```
